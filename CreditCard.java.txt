package CCard;public class CreditCard {    private String customerName;    private String cardNumber;    private double creditLimit;    private double currentBalance;    public CreditCard(String customerName, String cardNumber, double creditLimit) {        this.customerName = customerName;        this.cardNumber = cardNumber;        this.creditLimit = creditLimit;        this.currentBalance = creditLimit;    }    private boolean pmtApprove(double amount) {        return amount <= currentBalance;    }    protected boolean makePayment(double amount) {        if (pmtApprove(amount)) {            currentBalance -= amount;            return true;        } else {            System.out.println("The transaction is unsuccessful. The transaction balance exceeds the card balance.");            return false;        }    }    protected void printSummary() {        System.out.println("Customer name: " + customerName);        System.out.println("Card number: " + cardNumber);        System.out.println("Credit limit ($): " + (long)creditLimit);        System.out.println("Card balance ($): " + (long)currentBalance);    }    public String getCardNumber() {        return cardNumber;    }    public String getCustomerName() {        return customerName;    }    public long getCurrentBalanceAsLong() {        return (long) currentBalance;    }}
